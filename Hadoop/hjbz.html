环境搭建

1 配置hadoop 用户

[root@server1 ~]# useradd -u 1000 hadoop

[root@server1 ~]# id 1000

uid=1000(hadoop) gid=1000(hadoop) groups=1000(hadoop)

[root@server1 ~]# id hadoop

uid=1000(hadoop) gid=1000(hadoop) groups=1000(hadoop)

[root@server1 ~]#

2 下载并解压相关软件

[root@server1 ~]# su - hadoop

Last login: Sun Dec 30 22:09:47 CST 2018 on pts/0

[hadoop@server1 ~]$ ls

hadoop-2.8.5.tar.gz jdk-8u192-linux-x64.tar.gz

[hadoop@server1 ~]$ tar -xf hadoop-2.8.5.tar.gz

[hadoop@server1 ~]$ tar -xf jdk-8u192-linux-x64.tar.gz

[hadoop@server1 ~]$ ls

hadoop-2.8.5 hadoop-2.8.5.tar.gz jdk1.8.0_192 jdk-8u192-linux-x64.tar.gz

[hadoop@server1 ~]$ ln -s hadoop-2.8.5 hadoop

[hadoop@server1 ~]$ ln -s jdk.8.0_192 java

[hadoop@server1 ~]$ ll

total 428468

[hadoop@server1 ~]$

3 修改环境变量，使得hadoop运行在Java平台之上

[hadoop@server1 ~]$ source hadoop/etc/hadoop/hadoop-env.sh

export JAVA_HOME=/home/hadoop/java

[hadoop@server1 ~]$ source hadoop/etc/hadoop/hadoop-env.sh

4 修改java环境变量，使其可以查看到hadoop进程开启状态

[hadoop@server1 ~]$ vim .bash_profile

PATH=$PATH:$HOME/bin:/home/hadoop/java/bin

5 查看

[hadoop@server1 ~]$ source .bash_profile

[hadoop@server1 ~]$ jps

1950 jps
